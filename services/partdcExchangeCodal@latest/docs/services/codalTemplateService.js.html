let {

  BaseServiceClass
} = require('partdcFramework');

class CodalTemplateService extends BaseServiceClass {


  /**
   *Creates an instance of ExampleService.

  constructor(serviceName, serviceRedisKey, inputParam, serviceCallMethod,requestUrl) {
    super(serviceName, serviceRedisKey, inputParam, serviceCallMethod,requestUrl);
    
  }

  /**

  /**

   * @memberof exampleService
   */
  getTags(postfix = null) {
    let tags = super.getTags(postfix);

   *
   * در صورت بازنویسی حتما خط ذیل نوشته شود
   * 
   * this.logSystem.logClass.dlsTags = tags;
   * @memberof exampleService
   */
  async saveData() {
    let tags = this.getTags();
    this.makeTagByPrefix('tag1', tags);
    /**
     * کلیه فرایند های مرتبط
     */



    // این خط حتما در صورت بازنویسی نوشته شود
    this.logSystem.logClass.dlsTags = tags;
    await BaseConfig.partDataLayerInterface.insertData(tags, this.serviceData);
  }


}
