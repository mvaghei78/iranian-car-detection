/**
 * @class StatementsCodal

class StatementsCodal extends BaseServiceClass {


  /**
   *Creates an instance of ExampleService.
   * @param {!string} serviceName نام سرويس
   * @param {!string} serviceRedisKey نام منحصر به فرد سرويس جهت ذخيره سازي در رديس و عدم تکراري بودن فرايند اجراي ان
   * @param {object} inputParam پارامترهاي دريافتي از کاربر
   * @memberof  StatementsCodal
   */
  constructor(serviceName, serviceRedisKey, inputParam, serviceCallMethod) {
    super(serviceName, serviceRedisKey, inputParam, serviceCallMethod);
  }

  /**
   *@description بازنويسي تنظيمات مرتبط با رست
   * @returns {!object} تنظیمات درخواست رست
   * @memberof  StatementsCodal
   */
  getRestRequestOptions() {
    let options = super.getRestRequestOptions();
    let year = this.inputParam.year;
    let month = this.inputParam.month;
    let day = this.inputParam.day;
    options.url =  BaseConfig.restConfig.address + '/statements' +  `?year=${year}&month=${month}&day=${day}`;
    this.date = `${year}${month}${day}`;
    return options;
  }

    /**
   * @description جهت پردازش داده هاي دريافتي متد ذيل مي بايست پياده سازي شود
   *
   * @memberof  StatementsCodal
   * @returns {void}
   */
  async processData() {
    let serviceData = [];
    let data = this.serviceData;
    if (data.hasOwnProperty('Error')) {
      this.logSystem.logClass.codalError= data;
      throw Error(data);
    }

    else {
      let date = `${this.inputParam.year}${this.inputParam.month}${this.inputParam.day}`;
      let PreviousData = (await BaseConfig.mongoDBO.collection('CodalStatements').find({date: date}, {projection: {tracingNo: 1}}).toArray());
      //compare data with existing tracingNo on mongo and filter data 
      let filtered = data.filter(object => {
        return (PreviousData.map(item => {
          return item.tracingNo;
        })).indexOf(object.tracingNo) === -1;
      }
      );

      filtered.forEach((st) => {
        let newData = {};
        newData.data = st;
        newData.finglishSymbol = finglishIt(st.symbol);
        newData.date = st.publishDateTime.substring(0, 10).replace(/\//g, '');
        newData.letterType = st.letterType;
        newData.tracingNo = st.tracingNo;
        newData.period = st.period;
        serviceData.push(newData);
      });
      this.serviceData = serviceData;
    }

  /**
   *
   * بازنويسي فرايند ذخيره سازي داده
   *
   * اضافه کردن تگ به ليست تگ ها
   * @description
   *  در اين متد دو فرايند قابل بازنويسي است
   * فرايند افزودن تگ ها به ليست تگ ها
   *فرايند ذخيره سازي داده و ارتباط با دي ال اس
   *
   * در صورت بازنويسي حتما خط ذيل نوشته شود
   *
   * this.logSystem.logClass.dlsTags = tags;
   * @memberof  StatementsCodal
   * @returns {void}
   */
  async saveData() {

    let data = this.serviceData;
    if (data) {
      let defer = q.defer();
      try {
        async.eachSeries(data, async (item) => {

          // save data to mongo
          await BaseConfig.mongoDBO.collection('CodalStatements').updateOne(
            {tracingNo: item.tracingNo.toString()},
            {
              $set:
              item
            },
            {upsert: true});
        }, (error) => {
          if (error) {
            console.log('===============================');
            console.log('Error in innerSeries', error);
            console.log('===============================');
          }
          else
            defer.resolve();
        });
      }
      catch (e) {
        console.log(e);
        defer.reject(e);
      }
      return defer.promise;
    }

  }
}
